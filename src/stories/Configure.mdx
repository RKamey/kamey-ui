            >
              npm
            </button>
            <button

            >
              yarn
            </button>
            <button 
              className={`px-4 py-2 font-medium ${activeTab === 'bun' ? 'text-blue-500 border-b-2 border-blue-500' : 'text-gray-500 hover:text-gray-700'}`}

            >
              bun
            </button>
          </div>
          <div className="mt-2">

    ## Uso

    Para utilizar los componentes de kamey-components, se debe importar el componente deseado y utilizarlo en el código de la siguiente manera:

    ```jsx
    import { Button } from 'kamey-components';

    const App = () => {
      return (
        <Button>Click me!</Button>
      );
    }
    ```
